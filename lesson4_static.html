	$(document).ready(function(){
	var a = 0;
	var b = 0;
	var c = 0;
		$('.coefficient').keyup(function(){
			if($(this).hasClass('a')){
				a = parseFloat($(this).val());
			}
			if($(this).hasClass('b')){
				b = parseFloat($(this).val());
			}
			if($(this).hasClass('c')){
				c = parseFloat($(this).val());
			}
			if(isNan(a)){
				a=0;
			}
			if(isNan(b)){
				b=0;
			}
			if(isNan(c)){
				c=0;
			}
			if(!a){
				answer('Некорректное уравнение');
			}
			var d;
			d = b*b-4*a*c;
			if(d<0){
				answer('Корней нет');
				return;
			}
			if(d==0){
				var x = (-b)/(2*a);
				answer(x+' кратности 2');
			}else{
				var x1 = (-b-Math.sqrt(d))/(2*a);
				var x2 = (-b+Math.sqrt(d))/(2*a);
				answer(x1+'; '+x2);
			}
		});
	});
	function answer(text){
		$('#answer').html(text);
	}
