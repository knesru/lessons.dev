Еще одна простая программа. Checkboxes, debug

    set/get cookies 
    
	<div class="checkboxes">
		<input type="checkbox" id="checkall" />
	</div>


    
	$(document).ready(function() {
		for(var i=0; i<100; i++){
			$('.checkboxes')
	.append('<input type="checkbox" class="slaves" />');
		}
		$('#checkall').click(function(){
			if(!$(this).prop('checked')) {
				$('.slaves')
					.prop('checked', false);
			}else{
				$('.slaves')
					.prop('checked', true);
			}
		});
		$('.slaves').click(function(){
			if($('.slaves:checked').length>0){
				$('#checkall')
					.prop('checked', true);
			}
		});
	});
